class Solution:
    def longestConsecutive(self, nums: List[int]) -> int:
        map = {num: True for num in nums}

        longest = 0
        for num in map.keys():
            length = 1
            pre_num = num - 1
            while map.get(pre_num):
                length += 1
                pre_num -= 1
            longest = max(longest, length)
        return longest
```
